调用--“calendar-date”对应
preset:date为日期，datetime为日期-时间
invalid:设置不可用天数
minDate:设置最小时间
maxDate:设置最大时间

<input class="calendar-date" /> $(function () { var opt = { 'date': { preset: 'date', invalid: { daysOfWeek: [0, 6], daysOfMonth: ['5/1', '12/24', '12/25'] }, minDate: new Date(2012, 3, 10), maxDate: new Date(2015, 3, 10), }, 'datetime': { preset: 'datetime', invalid: { daysOfWeek: [0, 6], daysOfMonth: ['5/1', '12/24', '12/25'] }, minDate: new Date(2012, 3, 10, 9, 00), maxDate: new Date(2014, 7, 30, 15, 00), stepMinute: 5 }, } $('.calendar-date').scroller($.extend(opt['date'], { lang: 'zh', animate: 'flip' })); $('.calendar-datetime').scroller($.extend(opt['datetime'], { lang: 'zh', animate: 'flip' })); });
